36 void Rigidbody2D::UpdateMass(f32 newMass) {
37 mass = newMass;
38 if (type == BodyType::Static) {
39 inverseMass = 0.0f;
40 inverseInertia = 0.0f;
41 } else {
42 inverseMass = (mass > 0.0f) ? 1.0f / mass : 0.0f;
43 // Simple inertia approximation for a point mass
44 // This should be recalculated based on collider shape
45 inertia = mass;
46 inverseInertia = (inertia > 0.0f) ? 1.0f / inertia : 0.0f;
47 }
48 }

55 void Rigidbody2D::ApplyImpulse(const Vec2& impulse) {

60 void Rigidbody2D::ApplyImpulseAtPoint(const Vec2& impulse, const Vec2& contactPoint, const Vec2& centerOfMass) {
61 if (type != BodyType::Dynamic) return;
62
63 // Linear impulse
64 velocity += impulse * inverseMass;
65
66 // Angular impulse
67 if (!lockRotation) {
68 const Vec2 r = contactPoint - centerOfMass;
69 const auto cross = r.x * impulse.y - r.y * impulse.x;
70 angularVelocity += cross * inverseInertia;
71 }
72 }
73
74 void Rigidbody2D::ApplyTorque(f32 t) {
75 if (type != BodyType::Dynamic || lockRotation) return;
76 torque += t;
77 }

79 void Rigidbody2D::ClearForces() {
80 force = Vec2(0.0f);
81 torque = 0.0f;
82 }
