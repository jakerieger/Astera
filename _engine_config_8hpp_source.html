61 inline bool Load(const Path& filename) {
62 using namespace mINI;
63 const INIFile iniFile(filename);
64 INIStructure ini;
65 if (!iniFile.read(ini)) {
66 return false;
67 }
68
69 auto SplitBindings = [](const string& value) -> vector<string> {
70 vector<std::string> result;
71 std::stringstream ss(value);
72 string token;
73 while (std::getline(ss, token, ',')) {
74 result.push_back(token);
75 }
76 return result;
77 };
78
79 // Iterate Actions
80 for (auto const& [actionName, bindings] : ini["Actions"]) {
81 const auto split = SplitBindings(bindings);
82 ActionMap map;
83
84 for (const auto& binding : split) {
85 if (binding.find("Key.") != string::npos) {
86 const auto keyCodeStr = binding.substr(4, binding.length());
87 const auto keyCode = Input::kKeyCodeMap.at(keyCodeStr);
88 map.keys.push_back(keyCode);
89 } else if (binding.find("Mouse.") != string::npos) {
90 const auto mouseCodeStr = binding.substr(6, binding.length());
91 const auto mouseCode = Input::kMouseButtonCodeMap.at(mouseCodeStr);
92 map.mouseButtons.push_back(mouseCode);
93 }
94 }
95
96 actions[actionName] = map;
97 }
98
99 // TODO: Iterate Axes
100 for (auto const& [axisName, binding] : ini["Axes"]) {
101 // axisName = "Aim"
102 // binding = "Mouse.X"
103 }
104
105 return true;
106 }
107 };
