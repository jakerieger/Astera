149 void InputManager::UpdateMousePosition(const f64 x, const f64 y) {
150 if (!mEnabled) return;
151
152 mMouseDeltaX = CAST<f32>(x);
153 mMouseDeltaY = CAST<f32>(y);
154
155 constexpr f32 deadZone = 2.5f; // Might need to tweak this, also frame-rate dependent :(
156 if (std::abs(mMouseDeltaX) < deadZone) mMouseDeltaX = 0.0f;
157 if (std::abs(mMouseDeltaY) < deadZone) mMouseDeltaY = 0.0f;
158
159 mMouseX += x;
160 mMouseY += y;
161 }
