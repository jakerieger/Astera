302 template<>
303 N_ND inline u32 Color::To() const {
304 const auto redByte = CAST<u8>(FloatToU32(mRed));
305 const auto greenByte = CAST<u8>(FloatToU32(mGreen));
306 const auto blueByte = CAST<u8>(FloatToU32(mBlue));
307 const auto alphaByte = CAST<u8>(FloatToU32(mAlpha));
308 return (alphaByte << 24) | (redByte << 16) | (greenByte << 8) | blueByte;
309 }
310
313 template<>
314 N_ND inline string Color::To() const {
315 const u32 r = U32ToFloat(mRed + 0.5f);
316 const u32 g = U32ToFloat(mGreen + 0.5f);
317 const u32 b = U32ToFloat(mBlue + 0.5f);
318 std::stringstream ss;
319 ss << "#" << std::hex << std::setfill('0') << std::setw(2) << (r & 0xFF) << std::setw(2) << (g & 0xFF)
320 << std::setw(2) << (b & 0xFF);
321 return ss.str();
322 }

326 template<>
327 N_ND inline ImVec4 Color::To() const {
328 return ImVec4 {mRed, mGreen, mBlue, mAlpha};
